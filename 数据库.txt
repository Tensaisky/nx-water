CREATE TABLE `datatype` (
  `DATATYPE_ID` int(11) NOT NULL,
  `DATATYPE_NAME` varchar(45) NOT NULL,
  `DATATYPE_DESC` varchar(45) DEFAULT NULL,
  `DATATYPE_UNIT` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`DATATYPE_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `pgroup` (
  `PGROUP_ID` int(11) NOT NULL,
  `PGROUP_DESC` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`PGROUP_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `rmd` (
  `PGROUP_ID` int(11) NOT NULL,
  `RTU_ID` int(11) NOT NULL,
  `DATATYPE_ID` int(11) NOT NULL,
  `RMD_TIME` datetime NOT NULL,
  `RMD_VALUE` decimal(9,3) DEFAULT NULL,
  PRIMARY KEY (`RMD_TIME`,`DATATYPE_ID`,`PGROUP_ID`,`RTU_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `rtu` (
  `RTU_ID` int(11) NOT NULL,
  `RTU_PGROUP_ID` int(11) NOT NULL,
  `RTU_IS_VALID` int(11) DEFAULT NULL,
  `RTU_STATE` int(11) DEFAULT NULL,
  `RTU_DESC` varchar(45) DEFAULT NULL,
  `RTU_VER` int(11) DEFAULT NULL,
  `RTU_COM_TYPE` int(11) DEFAULT NULL,
  `RTU_ADD1` varchar(45) DEFAULT NULL,
  `RTU_ADD2` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`RTU_ID`,`RTU_PGROUP_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `user` (
  `USER_NAME` varchar(45) NOT NULL,
  `USER_DESC` varchar(45) DEFAULT NULL,
  `USER_PSW` varchar(45) DEFAULT NULL,
  `USER_ROLE` int(11) DEFAULT NULL,
  PRIMARY KEY (`USER_NAME`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
